```javascript
// src/utils/request.js
import axios from 'axios'
import { ElMessage } from 'element-plus'
import router from '@/router'

const service = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 10000,
  withCredentials: true // 允许携带Cookie
})

service.interceptors.request.use(
  config => {
    return config
  },
  error => Promise.reject(error)
)

service.interceptors.response.use(
  response => {
    const res = response.data
    if (res.code !== 200) {
      ElMessage.error(res.message || '请求失败')
      return Promise.reject(res)
    }
    return res
  },
  error => {
    if (error.response?.status === 401) {
      ElMessage.error('登录已过期')
      router.push('/login')
    } else {
      ElMessage.error(error.message || '网络错误')
    }
    return Promise.reject(error)
  }
)

export default service
```

### 3.4 任务服务层

```javascript
// services/taskService.js
const db = require('../config/database')

// 获取任务编号
const generateTaskNo = async () => {
  const date = new Date().toISOString().slice(0, 10).replace(/-/g, '')
  const [rows] = await db.query(
    'SELECT COUNT(*) as count FROM tasks WHERE DATE(created_at) = CURDATE()'
  )
  const seq = (rows[0].count + 1).toString().padStart(3, '0')
  return `RW-${date}-${seq}`
}

exports.getTasks = async (filters) => {
  let sql = `
    SELECT t.*, s.name as owner_name, p.period_name as period_name
    FROM tasks t
    LEFT JOIN staff s ON t.owner_id = s.staff_id
    LEFT JOIN periods p ON t.period_id = p.period_id
    WHERE 1=1
  `
  const params = []

  if (filters.period_id) {
    sql += ' AND t.period_id = ?'
    params.push(filters.period_id)
  }
  if (filters.owner_id) {
    sql += ' AND t.owner_id = ?'
    params.push(filters.owner_id)
  }
  if (filters.category) {
    sql += ' AND t.category = ?'
    params.push(filters.category)
  }
  if (filters.status) {
    sql += ' AND t.status = ?'
    params.push(filters.status)
  }
  if (filters.keyword) {
    sql += ' AND t.task_name LIKE ?'
    params.push(`%${filters.keyword}%`)
  }

  // 统计总数
  const [countResult] = await db.query(
    sql.replace('SELECT t.*, s.name as owner_name, p.period_name as period_name', 'SELECT COUNT(*) as total'),
    params
  )

  // 分页
  sql += ' ORDER BY t.end_date ASC LIMIT ? OFFSET ?'
  params.push(filters.page_size, (filters.page - 1) * filters.page_size)

  const [rows] = await db.query(sql, params)

  return {
    list: rows,
    pagination: {
      page: filters.page,
      page_size: filters.page_size,
      total: countResult[0].total
    }
  }
}

exports.createTask = async (data, createdBy) => {
  const taskNo = await generateTaskNo()
  const [result] = await db.query(
    `INSERT INTO tasks
     (task_no, task_name, period_id, owner_id, description, priority, category, start_date, end_date, progress, created_by)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [taskNo, data.task_name, data.period_id, data.owner_id, data.description,
     data.priority, data.category, data.start_date, data.end_date, 0, createdBy]
  )

  // 添加协助人
  if (data.collaborator_ids?.length) {
    const values = data.collaborator_ids.map(cid => [result.insertId, cid])
    await db.query(
      'INSERT INTO task_collaborators (task_id, staff_id) VALUES ?',
      [values]
    )
  }

  return result.insertId
}
```

### 4.1 任务编号生成算法

```javascript
/**
 * 生成任务编号
 * 格式: RW-YYYYMMDD-XXX
 * - RW: 任务前缀
 * - YYYYMMDD: 日期
 * - XXX: 当日序号，3位数字，不足补0
 */
const generateTaskNo = async () => {
  const date = new Date().toISOString().slice(0, 10).replace(/-/g, '')
  const [rows] = await db.query(
    'SELECT COUNT(*) as count FROM tasks WHERE DATE(created_at) = CURDATE()'
  )
  const seq = (rows[0].count + 1).toString().padStart(3, '0')
  return `RW-${date}-${seq}`
}
```

### 4.2 会议提醒计算算法

```javascript
/**
 * 计算会议提醒时间
 * @param {Date} startTime - 会议开始时间
 * @param {string} reminderSetting - 提醒设置
 * @returns {Date|null} - 提醒时间，如果不需要提醒返回null
 */
const calculateReminderTime = (startTime, reminderSetting) => {
  if (reminderSetting === 'none') return null

  const minutesMap = {
    '15min': 15,
    '30min': 30,
    '1hour': 60,
    '1day': 1440
  }

  const minutes = minutesMap[reminderSetting]
  if (!minutes) return null

  return new Date(new Date(startTime).getTime() - minutes * 60000)
}
```

```javascript
 * @returns {Object} - 分组后的待办
 */
const groupTodosByTime = (todos) => {
  const today = new Date()
  today.setHours(0, 0, 0, 0)

  const endOfWeek = new Date(today)
  endOfWeek.setDate(endOfWeek.getDate() + (7 - endOfWeek.getDay()))

  const groups = {
    today: [],
    thisWeek: [],
    future: [],
    noDate: [],
    completed: []
  }

  for (const todo of todos) {
    if (todo.status === 1) {
      groups.completed.push(todo)
      continue
    }

    if (!todo.deadline) {
      groups.noDate.push(todo)
      continue
    }

    const deadline = new Date(todo.deadline)
    deadline.setHours(0, 0, 0, 0)

    if (deadline.getTime() === today.getTime()) {
      groups.today.push(todo)
    } else if (deadline < endOfWeek) {
      groups.thisWeek.push(todo)
    } else {
      groups.future.push(todo)
    }
  }

  return groups
}
```
